                            // Reseteamos handlers

                            target._onSuccess = null; 

                            target._onFailure = null;

                            })

                            .catch(err => {

                                if(onFailure) onFailure(err);

                            });

        function sincronizarConPlanilla(callback = null) {

            if(APP_STATE.isSyncing) return;

                    if(data.relevar) APP_STATE.relevar.data = data.relevar.map((r, idx) => ({rowIndex: idx + 2, usuario: String(r[0]||"").trim(), numero: String(r[1]||"").trim(), sim: String(r[2]||"").trim(), estado: String(r[3]||"PENDIENTE")})).filter(r => r.usuario || r.numero || r.sim);

        function procesarDatosTabla(tableKey) {

            let info = APP_STATE[tableKey];

            let list = info.data;

            const q = info.search.toLowerCase();


            if (tableKey === 'altas') {

                if (q) list = list.filter(r => r.nombre.toLowerCase().includes(q) || r.correo.toLowerCase().includes(q) || r.sim.toLowerCase().includes(q) || r.telefono.includes(q));

                if (info.filterEstado) list = list.filter(r => r.estadoSnipe.toUpperCase().includes(info.filterEstado));

                if (info.filterTipo) list = list.filter(r => r.tipo.toUpperCase() === info.filterTipo);

            } else if (tableKey === 'bajas') {

                if (q) list = list.filter(r => r.linea.includes(q) || r.fechaBaja.includes(q));

            } else if (tableKey === 'relevar') {

                if (q) list = list.filter(r => r.usuario.toLowerCase().includes(q) || r.numero.includes(q) || r.sim.toLowerCase().includes(q));

            }


            const total = list.length;

            const totalPages = Math.ceil(total / info.limit) || 1;

            if (info.page > totalPages) info.page = totalPages;

            

            const start = (info.page - 1) * info.limit;

            const end = start + info.limit;

            const paginatedData = list.slice(start, end);


            return { data: paginatedData, total, totalPages, currentPage: info.page };

        }

        function parseDateStr(str) {

            if(!str) return null;

            let m = str.match(/(\d{2})\/(\d{2})\/(\d{4})/);

            if(m) return new Date(parseInt(m[3]), parseInt(m[2])-1, parseInt(m[1]));

            m = str.match(/(\d{4})-(\d{2})-(\d{2})/);

            if(m) return new Date(parseInt(m[1]), parseInt(m[2])-1, parseInt(m[3]));

            return null;

        }


        function actualizarEstadisticas() {

            const timeFilter = document.getElementById('filtro-tiempo').value;

            const now = new Date();

            

            const filterByTime = (dataArray, dateField) => {

                if(timeFilter === 'all') return dataArray;

                return dataArray.filter(item => {

                    const d = parseDateStr(item[dateField]);

                    if(!d) return true; 

                    

                    if(timeFilter === '30') {

                        const diffDays = Math.ceil(Math.abs(now - d) / (1000 * 60 * 60 * 24)); 

                        return diffDays <= 30;

                    }

                    if(timeFilter === 'month') {

                        return d.getMonth() === now.getMonth() && d.getFullYear() === now.getFullYear();

                    }

                    return true;

                });

            };

        function exportarCSV(tableKey) {

            let list = procesarDatosTabla(tableKey).data; 

            if(list.length === 0) return showToast("Aviso", "No hay datos para exportar en esta vista.", "error");

            

            let csvContent = "";

            let headers = [], rows = [];


            if(tableKey === 'altas') {

                headers = ["Nombre", "Correo", "Plan", "Tipo", "SIM", "Teléfono", "FechaCarga", "EstadoSnipe"];

                rows = list.map(r => [r.nombre, r.correo, r.plan, r.tipo, r.sim, r.telefono, r.fechaCarga, r.estadoSnipe]);

            } else if(tableKey === 'bajas') {

                headers = ["Linea", "FechaBaja", "Estado", "FechaCarga"];

                rows = list.map(r => [r.linea, r.fechaBaja, r.estado, r.fechaCarga]);

            } else {

                headers = ["Usuario", "Linea", "SIM", "Estado"];

                rows = list.map(r => [r.usuario, r.numero, r.sim, r.estado]);

            }


            csvContent += headers.join(";") + "\r\n";

            rows.forEach(rowArray => { let row = rowArray.map(item => `"${String(item).replace(/"/g, '""')}"`).join(";"); csvContent += row + "\r\n"; });


            const blob = new Blob(["\uFEFF"+csvContent], { type: 'text/csv;charset=utf-8;' }); 

            const url = URL.createObjectURL(blob);

            const link = document.createElement("a");

            link.setAttribute("href", url);

            link.setAttribute("download", `Export_${tableKey}_${new Date().toISOString().split('T')[0]}.csv`);

            document.body.appendChild(link); link.click(); document.body.removeChild(link);

            showToast("Descarga Lista", "El archivo CSV se descargó correctamente.");

        }
